Simple Centering

This window contains a Row layout container set to fill the window, and with the vertical and horizontal alignment centred.

Hello World! 

                fun main() = singleWindowApplication(
                    title = "Simple Centering"
                ) {
                    Row(
                        modifier = Modifier.fillMaxSize(),
                        horizontalArrangement = Arrangement.Center,
                        verticalAlignment = Alignment.CenterVertically
                    ) {
                        Text("Hello World!")
                    }
                }

Row Horizontal Alignment

This window contains a Column of Rows with various horizontal alignment settings.


                fun main() = singleWindowApplication(
                    title = "Horizontal Alignment",
                ) {
                    Column {
                        Row(
                            modifier = Modifier.fillMaxWidth(),
                            horizontalArrangement = Arrangement.Start
                        ) {
                            Button(...) { Text("A") }
                            Button(...) { Text("B") }
                            Button(...) { Text("C") }
                        }

                        Row(
                            modifier = Modifier.fillMaxWidth(),
                            horizontalArrangement = Arrangement.Center
                        ) {
                            Button(...) { Text("A") }
                            Button(...) { Text("B") }
                            Button(...) { Text("C") }
                        }

                        Row(
                            modifier = Modifier.fillMaxWidth(),
                            horizontalArrangement = Arrangement.End
                        ) {
                            Button(...) { Text("A") }
                            Button(...) { Text("B") }
                            Button(...) { Text("C") }
                        }
                    }
                }
